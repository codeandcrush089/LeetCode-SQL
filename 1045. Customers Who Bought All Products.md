```sql
SELECT
    customer_id
FROM
    Customer
GROUP BY
    customer_id  -- Group transactions by customer
HAVING
    -- Condition: The number of distinct products bought by the customer (A)
    -- must equal the total number of products available (B).
    COUNT(DISTINCT product_key) = (
        SELECT 
            COUNT(product_key) 
        FROM 
            Product
    );
```
