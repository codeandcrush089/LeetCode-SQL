```sql
-- Solution to find the IDs of dates where the temperature was higher than the previous day.

SELECT
    W1.id
FROM
    Weather W1,
    Weather W2
-- Self-join the table to compare today's (W1) temperature with yesterday's (W2) temperature.
WHERE
    -- 1. Check if W1's date is exactly one day after W2's date.
    -- DATEDIFF(day, date1, date2) calculates the difference in days.
    DATEDIFF(W1.recordDate, W2.recordDate) = 1 
    -- 2. Check if W1's temperature is higher than W2's temperature.
    AND W1.temperature > W2.temperature;
```
