```sql
SELECT
    -- Rename the date column as 'day' for the final output
    activity_date AS day,
    -- Count the number of unique user IDs for each day
    COUNT(DISTINCT user_id) AS active_users
FROM
    Activity
WHERE
    -- Filter 1: Activity must be on or before the end date
    activity_date <= '2019-07-27'
    AND
    -- Filter 2: Activity must be on or after the start date (30 days before 2019-07-27)
    activity_date > DATE_SUB('2019-07-27', INTERVAL 30 DAY)
    -- Alternative, simpler way to calculate the start date: activity_date >= '2019-06-28'
GROUP BY
    activity_date;
```
