```sql
-- Solution to report the distance traveled by each user.
-- Includes users who have no recorded rides, showing a distance of 0.

SELECT
    U.name,
    -- Calculate the total distance traveled by summing the distance from the Rides table.
    -- COALESCE is used to replace NULL (for users with no rides from the LEFT JOIN) with 0.
    COALESCE(SUM(R.distance), 0) AS travelled_distance
FROM
    Users AS U
-- Use a LEFT JOIN to ensure all users from the Users table are included, 
-- regardless of whether they have a matching entry in the Rides table.
LEFT JOIN
    Rides AS R ON U.id = R.user_id
-- Group the results by user ID and name to aggregate the total distance for each user.
GROUP BY
    U.id, U.name
ORDER BY
    -- Primary sorting criteria: distance in descending order
    travelled_distance DESC,
    -- Secondary sorting criteria: name in ascending order (alphabetical)
    U.name ASC;
```
