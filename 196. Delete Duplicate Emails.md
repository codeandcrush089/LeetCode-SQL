-- Solution to delete all duplicate emails, keeping only the entry with the smallest ID.

-- This solution uses a self-join to identify and delete rows (P1) that have a duplicate
-- email (P1.email = P2.email) but a larger ID (P1.id > P2.id) than the row being kept (P2).

DELETE P1
FROM 
    Person P1,
    Person P2
WHERE
    P1.email = P2.email AND P1.id > P2.id;
