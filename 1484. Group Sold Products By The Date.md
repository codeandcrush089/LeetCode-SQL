```sql
-- Solution to find for each date the number of different products sold and their names.

SELECT
    sell_date,
    -- 1. Count the number of distinct products sold on that date.
    COUNT(DISTINCT product) AS num_sold,
    -- 2. Concatenate the distinct product names into a single string, 
    --    sorted lexicographically and separated by a comma and a space.
    GROUP_CONCAT(
        DISTINCT product
        ORDER BY product ASC
        SEPARATOR ','
    ) AS products
FROM
    Activities
-- Group the results by sell_date to perform aggregation for each unique date.
GROUP BY
    sell_date
-- Order the final result table by sell_date.
ORDER BY
    sell_date;
```
