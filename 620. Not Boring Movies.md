```sql
SELECT
    id,
    movie,
    description,
    rating
FROM
    Cinema
WHERE
    -- Condition 1: id must be odd. The modulo operator (%) checks for a remainder of 1 when divided by 2.
    id % 2 = 1
    -- Condition 2: description must not be 'boring'.
    AND description != 'boring'
ORDER BY
    -- Order the results by rating from highest to lowest.
    rating DESC;
```
