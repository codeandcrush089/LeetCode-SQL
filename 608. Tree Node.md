```sql
SELECT
    id,
    CASE
        -- 1. Root Node: The parent ID (p_id) is NULL.
        WHEN p_id IS NULL THEN 'Root'
        -- 2. Leaf Node: The node's ID does NOT appear in the set of all parent IDs (p_id column).
        WHEN id NOT IN (SELECT DISTINCT p_id FROM Tree WHERE p_id IS NOT NULL) THEN 'Leaf'
        -- 3. Inner Node: If it's neither Root (p_id IS NOT NULL) nor Leaf (id IS in the p_id set).
        ELSE 'Inner'
    END AS type
FROM
    Tree
ORDER BY
    id;
```
