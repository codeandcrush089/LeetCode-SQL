```sql
SELECT
    D.name AS Department,
    E.name AS Employee,
    E.salary AS Salary
FROM
    Employee E
JOIN
    Department D ON E.departmentId = D.id
WHERE
    -- Use a Common Table Expression (CTE) or subquery to rank salaries
    (E.departmentId, E.salary) IN (
        SELECT
            departmentId,
            MAX(salary)
        FROM
            Employee
        GROUP BY
            departmentId
    )
ORDER BY
    Department, Salary DESC;
```
