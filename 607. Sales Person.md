```sql
SELECT
    s.name
FROM
    SalesPerson AS s
WHERE
    -- Select salespersons whose ID is NOT in the list of IDs 
    -- who placed orders for the company named 'RED'.
    s.sales_id NOT IN (
        SELECT
            o.sales_id
        FROM
            Orders AS o
        JOIN
            Company AS c ON o.com_id = c.com_id
        WHERE
            c.name = 'RED'
    );
```
