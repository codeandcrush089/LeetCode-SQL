```sql
-- Solution to report the Capital gain/loss for each stock.
-- The total capital gain/loss is calculated by summing all 'Sell' prices and 
-- subtracting the sum of all 'Buy' prices for each distinct stock.

SELECT
    stock_name,
    -- We use conditional aggregation (CASE inside SUM) to treat 'Buy' operations 
    -- as a negative cost (subtraction) and 'Sell' operations as a positive gain (addition).
    SUM(
        CASE
            WHEN operation = 'Sell' THEN price
            WHEN operation = 'Buy' THEN -price
            -- Since the problem guarantees that all operations are 'Buy' or 'Sell', 
            -- the ELSE case is technically not needed but is good practice.
            ELSE 0 
        END
    ) AS capital_gain_loss
FROM
    Stocks
GROUP BY
    stock_name
ORDER BY
    stock_name; -- Added ORDER BY for consistent output, though not strictly required by the prompt
```
